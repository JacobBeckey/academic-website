\newex{1}{6} \textbf{Trace}

\begin{enumerate}[(i)]
    \item Prove that the trace is linear (i.e $\tr{[A+B]}=\tr{[A]}+\tr{[B]}$ and $\tr{[zA]}=z\tr{[A]}$).
	\item Show that the space of traceless matrices, $\lbrace A\in M_n(\mathbb{F})\colon \tr(A) = 0\rbrace$, is a subspace of $M_n(\mathbb{F})$. What is its dimension?
	\item Suppose $V$ is an inner product space and $T \in \mathcal{L}(V)$. Prove that $\tr{T^{*}}=\overline{\tr{T}}$. \\\textit{Hint: use Axler 8.55 in the fourth edition.}
\end{enumerate}

\bigskip\newex{2}{9} \textbf{Determinant}

\begin{enumerate}[(i)]
	\item Prove that $\det{(AB)}=\det{(A)}\det{(B)}$ for all $2$-by-$2$ matrices $A,B$. 
    \item Consider the matrix 
    \begin{align}
        \begin{pmatrix}
            6 & -1 \\
            2 & 3
        \end{pmatrix}.
    \end{align}
    Solve the equation $\det{(A-\lambda I)} =0$ to find the eigenvalues of $A$.
    \item Let $\lambda_1, \lambda_2$ be two eigenvalues of a matrix $A$. Define the mean and the product of the eigenvalues of a 2-by-2 matrix as
    \begin{align}
        m &= \frac{1}{2}\tr{[A]}=\frac{1}{2}(\lambda_1 + \lambda_2),\\
        p &= \det{(A)} = \lambda_1 \lambda_2.
    \end{align}
    Prove that the eigenvalues of $A$ are given as 
    \begin{align}
        \lambda_{1}, \lambda_2 &= m \pm \sqrt{m^2-p}.
    \end{align}
\end{enumerate}
